//Relation Operators
#include <stdio.h>
int main()
{
	int n;
	printf("Enter the value of n:\n");
	scanf("%d",&n);
	printf("\n(n<10) is =%d\n",(n<10));  /* 1=true and 0= fals*/
	printf("(n>10) is =%d\n",(n>10));
	printf("(n<=10) is =%d\n",(n<=10));
	printf("(n>=10) is =%d\n",(n>=10));
	printf("(n == 10) is =%d\n",(n == 10));
	printf("(n != 10) is =%d\n",(n != 10));
	return 0;
}
