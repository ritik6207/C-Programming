//program to explain operator Precedence
#include <stdio.h>

int main()
{
    float result;
    result= 4.0 + 3.0*8.0/4.0 ;
    printf("\n %f",result);
    result=1.0/2.0 + 3.0;
    printf("\n %f",result);
    result=(1.0 + 2.0)/3.0;
    printf("\n %f",result);
    result=(1.0 + 2.0 / 3.0) + 4.0;
    printf("\n %f",result);
    return 0;
}
